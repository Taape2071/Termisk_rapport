import numpy as np
import matplotlib.pyplot as plt


def fleas(N, t):
    time = np.linspace(0, t, t)
    jumps = np.random.randint(0, N, t)


    NA = np.ones(N)
    NB = np.zeros(N)
    first_A = []
    first_B = []


    for i in (jumps):
        if NA[i] == 1:
            NA[i] = 0
            NB[i] = 1
        else:
            NA[i] = 1
            NB[i] = 0
        
        first_A.append(np.sum(NA)/N)
        first_B.append(np.sum(NB)/N)
        first_A[0] = 1
        first_B[0] = 0

    return first_A, first_B, time

def analytical(N, t):
    t = np.linspace(0, t, t)
    return (N/2)*(1 + np.e**(-2*t*1/N))/N, t

#This plots the distribution of # fleas on dog A
ans1 = fleas(1000, 5000)
plt.title("Distribution dog A, N = 1000")
plt.plot(ans1[2], ans1[0], label = "Numerical")

ans3 = analytical(1000, 5000)
plt.plot(ans3[1], ans3[0], label = "Analytical")
plt.legend()
plt.ylabel("Part fleas normed")
plt.xlabel("Time")
plt.show()
